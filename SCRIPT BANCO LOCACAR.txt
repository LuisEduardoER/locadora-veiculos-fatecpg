CREATE TABLE TB_USUARIO(
ID INTEGER NOT NULL AUTO_INCREMENT,
NOME VARCHAR(200),
APELIDO VARCHAR(100),
EMAIL VARCHAR(150),
SENHA VARCHAR(50),
PRIMARY KEY (ID)
);

CREATE TABLE TB_MODULO (
ID INTEGER NOT NULL AUTO_INCREMENT,
NOME  VARCHAR(200),
URL  VARCHAR(250),
ICONE  VARCHAR(100),
PRIMARY KEY(ID)
);

CREATE TABLE TB_CLIENTE(
ID INTEGER NOT NULL AUTO_INCREMENT,
NOME  VARCHAR(200),
CPF_CNPJ  VARCHAR(20),
EMAIL  VARCHAR(150),
TELEFONE VARCHAR(10),
CELULAR VARCHAR(11),
ENDERECO VARCHAR(500),
CEP VARCHAR(8),
PRIMARY KEY(ID)
);

CREATE TABLE TB_MARCA(
ID INTEGER NOT NULL AUTO_INCREMENT,
NOME VARCHAR(200),
PRIMARY KEY(ID)
);

CREATE TABLE TB_COR(
ID INTEGER NOT NULL AUTO_INCREMENT,
NOME VARCHAR(50),
PRIMARY KEY(ID)
);

CREATE TABLE TB_PERFIL(
ID INTEGER NOT NULL AUTO_INCREMENT,
MODULOID INTEGER NOT NULL,
USUARIOID INTEGER NOT NULL,
PRIMARY KEY(ID),
FOREIGN KEY(MODULOID) REFERENCES TB_MODULO(ID),
FOREIGN KEY(USUARIOID) REFERENCES TB_USUARIO(ID)
);



CREATE TABLE TB_MODELO(
ID INTEGER NOT NULL AUTO_INCREMENT,
MARCAID INTEGER NOT NULL,
NOME VARCHAR(100),
PRIMARY KEY (ID),
FOREIGN KEY(MARCAID) REFERENCES TB_MARCA(ID)
);

CREATE TABLE TB_CARRO(
ID INTEGER NOT NULL AUTO_INCREMENT,
CORID INTEGER NOT NULL,
MODELOID INTEGER NOT NULL,
PLACA VARCHAR(7),
CHASSI VARCHAR(50),
DESCRICAO VARCHAR(8000),
DIARIA DECIMAL(18,2),
IMG VARCHAR(250),
DISPONIVEL BIT,
PRIMARY KEY (ID),
FOREIGN KEY (MODELOID) REFERENCES TB_MODELO(ID),
FOREIGN KEY (CORID) REFERENCES TB_COR(ID)
);

CREATE TABLE LOCACAO(
ID INTEGER NOT NULL AUTO_INCREMENT,
CARROID INTEGER NOT NULL,
CLIENTEID INTEGER NOT NULL,
DATA_INICIO VARCHAR(20)
DATA_ENTREGA VARCHAR(20),
PRIMARY KEY (ID),
FOREIGN KEY(CARROID)REFERENCES TB_CARRO(ID),
FOREIGN KEY(CLIENTEID)REFERENCES TB_CLIENTE(ID) 
)
